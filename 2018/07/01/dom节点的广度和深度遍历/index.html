dom节点的广度和深度遍历

By Chenhw

发表于 2018-07-01

文章目录

JS又重新引入了元素树的概念。这个的话，在我们实际应用中，用的比较普遍。
元素树：仅仅包含元素节点的树结构，不是一颗新树，尽是节点数的子集；
treeNode.parentElement 找节点的父元素；
treeNode.children返回节点的所有子元素；
treeNode.firstElementChild 第一个直接子元素；
treeNode.lastElementChild 最后一个直接字元素；
treeNode.previousElementSibling 找treeNode的前一个兄弟元素；
treeNode.nextElementSibling 找treeNode的下个兄弟元素；

js代码

var div = document.getElementById('div1');
var treeNode = document.createTreeWalker(div, NodeFilter.SHOW_ELEMENT, null, false);

//console.log(treeNode.firstChild().tagName);
//treeNode.firstChild();  //转到<div>
//treeNode.nextSibling();  //转到<ul>

var node = treeNode.firstChild();  //转到第一个<li>
var nodes = [];
while (node !== null) {//真正没有节点了，就结束遍历
    nodes.push(node.tagName.toLowerCase());
    console.log(node.tagName.toLowerCase());
    if(node.nextSibling!==null){//如果可以进入一层，就进入
        node = treeNode.firstChild();
    }
    node = treeNode.nextSibling();//在同一层进行横向遍历
    if(node === null){//如果为空则则换到下一层
        treeNode.parentNode();
        node = treeNode.nextSibling();
    }
}
console.log(nodes);
