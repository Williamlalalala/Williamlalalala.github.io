使用js来修改和控制cookie的内容

By Chenhw

Published 2017-11-18

Contents 
<html>
<head>
<script type="text/javascript">
    function getCookie(c_name)
    {
            if(document.cookie.length>0)
            {
                var c_start=document.cookie.indexOf(c_name+"=");
                var c_end;
                if(c_start!=-1)
                {
                    c_start=c_start+c_name.length+1;
                    c_end=document.cookie.indexOf(";",c_start);
                if(c_end==-1){
                    c_end=document.cookie.length;
                }
                return unescape(document.cookie.substring(c_start,c_end));
                }
            }
            return "";
    }
    function setCookie(c_name,value,expiredays)
    {
        var exdate=new Date();
        exdate.setDate(exdate.getDate()+expiredays);
        document.cookie=c_name+"="+escape(value)+((expiredays==null)?"":";expires="+exdate.toGMTString());
    }
    function checkCookie()
    {
        var username=getCookie('username');
        if(username!=null&&username!="")
        {
            alert('Welcome again'+username+'!');
        }
        else
        {
            username=prompt('Please enter yourname:',"");
            if(username!=null&&username!="")
            {
                setCookie('username',username,365);
            }
        }
    }
    window.onload=function(){
        checkCookie();
    }
</script>
</head>
<body>
</body>
</html>

这里解释一波，开始的那个function中我使用的checkcookie函数在开始就调用了getcookie函数，如果返回的是空值，那么直接会调用prompt()函数，这个函数和alert是差不多的一个弹出框，但是这里会有一个框和一个提交按钮，在这里输入的值会返回到我们的js中，然后我们把这个值记录下来到下面用到。然后就是setCookie这个函数，我们在这里使用了js自带的一个new Date（）函数，用来获取时间，是为了设置expire的值然后我们把这个时间设为在本地获取的时间加上了356，即为传进来的值，然后整个cookie里的值通过document.cookie进行设置最终的字符串为：
c_name+”=”+escape(value)+((expiredays==null)?””:”;expires=”+exdate.toGMTString());
/即为/username=’chenhw’;expires=’Sat Nov 18 2018 12:39:11 GMT-0800 (太平洋标准时间)’
((expiredays==null)?””:”;expires=”这里的语法是如果刚才设置的expiredays（就是你设置的cookie的存在时间）不为空，就执行字符串”;expires”。
escape(value)就是把value这个字符串给转换成十六进制的转义序列
